9 // Module Bufferization is an extension of Comprehensive Bufferize that
10 // bufferizes function boundaries. It provides `BufferizableOpInterface`
11 // implementations for FuncOp, CallOp and ReturnOp.
12 //
13 // Module Bufferization is run via `runModuleBufferize(ModuleOp, ...)`. This
14 // function analyzes the given module and determines the order of analysis and
15 // bufferization: Functions that are called are processed before their
16 // respective callers.
17 //
18 // After analyzing a FuncOp, additional information about its bbArgs is
19 // gathered through PostAnalysisStepFns and stored in `FuncAnalysisState`.
20 //
21 // * `aliasingFuncOpBBArgsAnalysis` determines the equivalent/aliasing bbArgs
22 // for
23 // each tensor return value (if any).
24 // * `funcOpBbArgReadWriteAnalysis` determines whether or not a tensor bbArg is
25 // read/written.
26 //
27 // Only tensors that are equivalent to some FuncOp bbArg may be returned.
28 // Bufferization currently fails if other tensors (in particular tensors that
29 // bufferize out-of-place and result in a new buffer allocation) are returned.
30 // In the future, such allocations could be hoisted to the caller.
31 //
32 // Example: `foo` fails bufferization because %0 is not equivalent to any bbArg.
33 // ```
34 // func @foo() -> tensor<?xf32> {
35 // %0 = linalg.init_tensor [...] : tensor<?xf32>
36 // return %0 : tensor<?xf32>
37 // }
38 // ```
39 //
40 // Module Bufferization implements the following calling convention.
41 //
42 // * In the absence of conflicts within a FuncOp, the FuncOp's bbArgs may always
43 // be written to in-place.
44 // * If a tensor operand of a CallOp is read after the CallOp, the operand of
45 // the CallOp must bufferize out-of-place.
46 //
47 // Example: The tensor.insert op bufferizes in-place because it is allowed to
48 // modify the buffer of `%t1` directly. The CallOp in `caller` must bufferize
49 // out-of-place because `%t0` is modified by the callee but read by the
50 // tensor.extract op. The analysis of CallOps decides whether an OpOperand must
51 // bufferize out-of-place based on results of `funcOpBbArgReadWriteAnalysis`.
52 // ```
53 // func @callee(%t1 : tensor<?xf32>) -> tensor<?xf32> {
54 // %f = ... : f32
55 // %0 = tensor.insert %f into %t1[...] : tensor<?xf32>
56 // return %0 : tensor<?xf32>
57 // }
58 //
59 // func @caller() -> () {
60 // %t0 = ... : tensor<?xf32>
61 // %1 = call @callee(%t0) : (tensor<?xf32>) -> (tensor<?xf32>)
62 // %2 = tensor.extract %1[...] : tensor<?xf32>
63 // }
64 // ```
65 //
66 // Note: If a function is external, `funcOpBbArgReadWriteAnalysis` cannot
67 // analyze the function body. In such a case, the CallOp analysis conservatively
68 // assumes that each tensor OpOperand is both read and written.
69 //
70 // TODO: Add FuncOp attributes so that bbArgs of external FuncOps can be marked
71 // as "not reading" and/or "not writing".

96 /// Extra analysis state that is required for bufferization of function
97 /// boundaries.
98 struct FuncAnalysisState : public DialectAnalysisState {
99  // Note: Function arguments and/or function return values may disappear during
100  // bufferization. Functions and their CallOps are analyzed and bufferized
101  // separately. To ensure that a CallOp analysis/bufferization can access an
102  // already bufferized function's analysis results, we store bbArg/return value
103  // indices instead of BlockArguments/OpOperand pointers.

105  /// A set of block argument indices.
106  using BbArgIndexSet = DenseSet<int64_t>;
107 
108  /// A mapping of indices to indices.
109  using IndexMapping = DenseMap<int64_t, int64_t>;
110 
111  /// A mapping of indices to a list of indices.
112  using IndexToIndexListMapping = DenseMap<int64_t, SmallVector<int64_t>>;
113 
114  /// A mapping of ReturnOp OpOperand indices to equivalent FuncOp BBArg
115  /// indices.
116  DenseMap<func::FuncOp, IndexMapping> equivalentFuncArgs;
117 
118  /// A mapping of ReturnOp OpOperand indices to aliasing FuncOp BBArg indices.
119  DenseMap<func::FuncOp, IndexToIndexListMapping> aliasingFuncArgs;
120 
121  /// A mapping of FuncOp BBArg indices to aliasing ReturnOp OpOperand indices.
122  DenseMap<func::FuncOp, IndexToIndexListMapping> aliasingReturnVals;
123 
124  /// A set of all read BlockArguments of FuncOps.
125  DenseMap<func::FuncOp, BbArgIndexSet> readBbArgs;
126 
127  /// A set of all written-to BlockArguments of FuncOps.
128  DenseMap<func::FuncOp, BbArgIndexSet> writtenBbArgs;
129 
130  /// Keep track of which FuncOps are fully analyzed or currently being
131  /// analyzed.
132  DenseMap<func::FuncOp, FuncOpAnalysisState> analyzedFuncOps;
133 
134  /// This function is called right before analyzing the given FuncOp. It
135  /// initializes the data structures for the FuncOp in this state object.
136  void startFunctionAnalysis(func::FuncOp funcOp) {
137  analyzedFuncOps[funcOp] = FuncOpAnalysisState::InProgress;
138  auto createdEquiv = equivalentFuncArgs.try_emplace(funcOp, IndexMapping());
139  auto createdAliasingOperands =
140  aliasingFuncArgs.try_emplace(funcOp, IndexToIndexListMapping());
141  auto createdAliasingResults =
142  aliasingReturnVals.try_emplace(funcOp, IndexToIndexListMapping());
143  auto createdRead = readBbArgs.try_emplace(funcOp, BbArgIndexSet());
144  auto createdWritten = writtenBbArgs.try_emplace(funcOp, BbArgIndexSet());
145  (void)createdEquiv;
146  (void)createdAliasingOperands;
147  (void)createdAliasingResults;
148  (void)createdRead;
149  (void)createdWritten;
150 #ifndef NDEBUG
151  assert(createdEquiv.second && "equivalence info exists already");
152  assert(createdAliasingOperands.second && "aliasing info exists already");
153  assert(createdAliasingResults.second && "aliasing info exists already");
154  assert(createdRead.second && "bbarg access info exists already");
155  assert(createdWritten.second && "bbarg access info exists already");
156 #endif // NDEBUG
157  }
158 };

224 /// Store function BlockArguments that are equivalent to/aliasing a returned
225 /// value in FuncAnalysisState.
226 static LogicalResult
227 aliasingFuncOpBBArgsAnalysis(Operation *op, AnalysisState &state,
228  BufferizationAliasInfo &aliasInfo,
229  SmallVector<Operation *> &newOps) {
230  FuncAnalysisState &funcState = getFuncAnalysisState(state);
231 
232  // Support only single return-terminated block in the function.
233  auto funcOp = cast<func::FuncOp>(op);
234  func::ReturnOp returnOp = getAssumedUniqueReturnOp(funcOp);
235  assert(returnOp && "expected func with single return op");
236 
237  for (OpOperand &returnVal : returnOp->getOpOperands())
238  if (returnVal.get().getType().isa<RankedTensorType>())
239  for (BlockArgument bbArg : funcOp.getArguments())
240  if (bbArg.getType().isa<RankedTensorType>()) {
241  int64_t returnIdx = returnVal.getOperandNumber();
242  int64_t bbArgIdx = bbArg.getArgNumber();
243  if (aliasInfo.areEquivalentBufferizedValues(returnVal.get(), bbArg)) {
244  funcState.equivalentFuncArgs[funcOp][returnIdx] = bbArgIdx;
245  if (state.getOptions().testAnalysisOnly)
246  annotateEquivalentReturnBbArg(returnVal, bbArg);
247  }
248  if (aliasInfo.areAliasingBufferizedValues(returnVal.get(), bbArg)) {
249  funcState.aliasingFuncArgs[funcOp][returnIdx].push_back(bbArgIdx);
250  funcState.aliasingReturnVals[funcOp][bbArgIdx].push_back(returnIdx);
251  }
252  }
253 
254  return success();
255 }

257 /// Return true if the buffer of the given tensor value is written to. Must not
258 /// be called for values inside not yet analyzed functions. (Post-analysis
259 /// steps do not have to be run yet, i.e., "in progress" is also OK.)
260 static bool isValueWritten(Value value, const AnalysisState &state,
261  const BufferizationAliasInfo &aliasInfo) {
262 #ifndef NDEBUG
263  assert(value.getType().isa<TensorType>() && "expected TensorType");
264  func::FuncOp funcOp;
265  if (auto bbArg = value.dyn_cast<BlockArgument>()) {
266  Operation *owner = bbArg.getOwner()->getParentOp();
267  funcOp = isa<func::FuncOp>(owner) ? cast<func::FuncOp>(owner)
268  : owner->getParentOfType<func::FuncOp>();
269  } else {
270  funcOp = value.getDefiningOp()->getParentOfType<func::FuncOp>();
271  }
272  assert(getFuncOpAnalysisState(state, funcOp) !=
273  FuncOpAnalysisState::NotAnalyzed &&
274  "FuncOp must be fully analyzed or analysis in progress");
275 #endif // NDEBUG
276 
277  bool isWritten = false;
278  aliasInfo.applyOnAliases(value, [&](Value val) {
279  for (OpOperand &use : val.getUses())
280  if (state.isInPlace(use) && state.bufferizesToMemoryWrite(use))
281  isWritten = true;
282  });
283  return isWritten;
284 }

287  bool isRead, bool isWritten) {

302 /// Determine which FuncOp bbArgs are read and which are written. If this
303 /// PostAnalysisStepFn is run on a function with unknown ops, it will
304 /// conservatively assume that such ops bufferize to a read + write.
305 static LogicalResult
306 funcOpBbArgReadWriteAnalysis(Operation *op, AnalysisState &state,
307  BufferizationAliasInfo &aliasInfo,
308  SmallVector<Operation *> &newOps) {
309  FuncAnalysisState &funcState = getFuncAnalysisState(state);
310  auto funcOp = cast<func::FuncOp>(op);
311 
312  // If the function has no body, conservatively assume that all args are
313  // read + written.
314  if (funcOp.getBody().empty()) {
315  for (BlockArgument bbArg : funcOp.getArguments()) {
316  funcState.readBbArgs[funcOp].insert(bbArg.getArgNumber());
317  funcState.writtenBbArgs[funcOp].insert(bbArg.getArgNumber());
318  }
319 
320  return success();
321  }
322 
323  for (BlockArgument bbArg : funcOp.getArguments()) {
324  if (!bbArg.getType().isa<TensorType>())
325  continue;
326  bool isRead = state.isValueRead(bbArg);
327  bool isWritten = isValueWritten(bbArg, state, aliasInfo);
328  if (state.getOptions().testAnalysisOnly)
329  annotateFuncArgAccess(funcOp, bbArg, isRead, isWritten);
330  if (isRead)
331  funcState.readBbArgs[funcOp].insert(bbArg.getArgNumber());
332  if (isWritten)
333  funcState.writtenBbArgs[funcOp].insert(bbArg.getArgNumber());
334  }
335 
336  return success();
337 }

370 /// tensor is replaced by the corresponding buffer type.
371 /// In order for all the callers to agree, this *must* bufferize to the most
372 /// dynamic buffer type supported.
373 /// A later pass across all CallOps in the module can decide whether to simplify
374 /// the types of to version according to some cost model.
375 static FunctionType
376 getBufferizedFunctionType(MLIRContext *ctx, TypeRange argumentTypes,
377  TypeRange resultTypes,
378  const BufferizationOptions &options) {
379  auto rewrite = [&](Type t) -> Type {
380  // TODO: non-zero address space.
381  // TODO: layout information if relevant.
382  if (auto tensorType = t.dyn_cast<TensorType>())
383  return getMemRefType(tensorType, options);
384  return t;
385  };
386  auto argTypes = llvm::to_vector<4>(llvm::map_range(argumentTypes, rewrite));
387  auto retTypes = llvm::to_vector<4>(llvm::map_range(resultTypes, rewrite));
388  return FunctionType::get(ctx, argTypes, retTypes);

391 /// Gather equivalence info of CallOps.
392 /// Note: This only adds new equivalence info if the called function was already
393 /// analyzed.
394 // TODO: This does not handle cyclic function call graphs etc.
395 static void equivalenceAnalysis(func::FuncOp funcOp,
396  BufferizationAliasInfo &aliasInfo,
397  FuncAnalysisState &funcState) {
398  funcOp->walk([&](func::CallOp callOp) {
399  func::FuncOp calledFunction = getCalledFunction(callOp);
400  assert(calledFunction && "could not retrieved called func::FuncOp");
401 
402  // No equivalence info available for the called function.
403  if (!funcState.equivalentFuncArgs.count(calledFunction))
404  return WalkResult::skip();
405 
406  for (auto it : funcState.equivalentFuncArgs[calledFunction]) {
407  int64_t returnIdx = it.first;
408  int64_t bbargIdx = it.second;
409  Value returnVal = callOp.getResult(returnIdx);
410  Value argVal = callOp->getOperand(bbargIdx);
411  aliasInfo.unionEquivalenceClasses(returnVal, argVal);
412  }
413 
414  return WalkResult::advance();
415  });
416 }

418 /// Rewrite the `funcOp` arguments analysis return values and terminator into
419 /// buffer form (using the canonical memref layout for now), according to the
420 /// inPlace-bufferizable information of the function arguments.
421 ///
422 /// This relies on a buffer equivalence analysis of each return operand. When a
423 /// result buffer is equivalent to a BlockArgument of `funcOp`, it can be
424 /// dropped from the return values and becomes inplaceable at all callers. This
425 /// assumes all CallOp perform the necessary work to clone operands so as to
426 /// make them inplaceable. Reliance on this logic will need to be relaxed in the
427 /// future.
428 ///
429 /// Note: Returning a memref currently fails bufferization. If such memrefs
430 /// originate from an op with an Alloc effect, they could be hoisted in the
431 /// future.
432 static LogicalResult bufferizeFuncOpBoundary(func::FuncOp funcOp,
433  RewriterBase &rewriter,
434  BufferizationState &state) {
435  const FuncAnalysisState &funcState =
436  getFuncAnalysisState(state.getAnalysisState());
437 
438  // If nothing to do then we are done.
439  if (!llvm::any_of(funcOp.getFunctionType().getInputs(), isaTensor) &&
440  !llvm::any_of(funcOp.getFunctionType().getResults(), isaTensor))
441  return success();
442 
443  // Get the bufferized FunctionType for funcOp or construct it if not yet
444  // available.
445  // TODO: Atm we have 3 cases:
446  // 1. if a function is called from within the Module, it must have bufferized
447  // to inplaceable tensor results.
448  // 2. if it is bodiless, it must have bufferized and is not allowed to have
449  // result tensors.
450  // 3. if it is not called internally, it still must bufferize to inplaceable
451  // tensor results and we construct it now (e.g. top-level function called
452  // externally).
453  // -> Figure out a better layering.
454  TypeRange resultTypes;
455 
456  // Corner case: Bodiless FuncOp
457  // ============================
458  // The body of such functions is assumed opaque and we can't know the
459  // bufferization contract they want to enforce atm.
460  // As a consequence, only support functions that don't return any tensor atm.
461  if (funcOp.getBody().empty()) {
462  if (llvm::any_of(funcOp.getFunctionType().getResults(), isaTensor))
463  return funcOp->emitError() << "cannot bufferize bodiless function that "
464  << "returns a tensor";
465  FunctionType bufferizedFuncType = getBufferizedFunctionType(
466  funcOp.getContext(), funcOp.getFunctionType().getInputs(),
467  funcOp.getFunctionType().getResults(), state.getOptions());
468  funcOp.setType(bufferizedFuncType);
469  return success();
470  }
471 
472  // Support only single return-terminated block in the function.
473  func::ReturnOp returnOp = getAssumedUniqueReturnOp(funcOp);
474  assert(returnOp && "expected func with single return op");
475 
476  // 1. For each FuncOp result, keep track of which inplace argument it reuses.
477  SmallVector<Value> returnValues;
478  for (OpOperand &returnOperand : returnOp->getOpOperands()) {
479  Value returnVal = returnOperand.get();
480 
481  // If not a renturn tensor type just forward it.
482  if (!returnVal.getType().isa<RankedTensorType>()) {
483  returnValues.push_back(returnVal);
484  continue;
485  }
486 
487  // If return operand is equivalent to some bbArg, no need to return it.
488  auto funcOpIt = funcState.equivalentFuncArgs.find(funcOp);
489  if (funcOpIt != funcState.equivalentFuncArgs.end() &&
490  funcOpIt->second.count(returnOperand.getOperandNumber()))
491  continue;
492 
493  // Cast values at the call site if necessary.
494  returnValues.push_back(
495  getNonCastedValue(*state.getBuffer(rewriter, returnOperand)));
496  }
497 
498  // 2. Rewrite the terminator without the inPlace bufferizable values.
499  ValueRange retValues{returnValues};
500  FunctionType bufferizedFuncType = getBufferizedFunctionType(
501  funcOp.getContext(), funcOp.getFunctionType().getInputs(),
502  retValues.getTypes(), state.getOptions());
503  OpBuilder b(returnOp);
504  b.create<func::ReturnOp>(returnOp.getLoc(), returnValues);
505  returnOp->erase();
506 
507  // 3. Rewrite the bbArgs.
508  // Iterate on the original `numArgs` and replace them in order.
509  // This guarantees the argument order still matches after the rewrite.
510  Block &frontBlock = funcOp.getBody().front();
511  unsigned numArgs = frontBlock.getNumArguments();
512  for (unsigned idx = 0; idx < numArgs; ++idx) {
513  auto bbArg = frontBlock.getArgument(0);
514  auto tensorType = bbArg.getType().dyn_cast<TensorType>();
515  // Non-tensor types are just forwarded.
516  if (!tensorType) {
517  frontBlock.addArgument(bbArg.getType(), bbArg.getLoc());
518  bbArg.replaceAllUsesWith(frontBlock.getArguments().back());
519  frontBlock.eraseArgument(0);
520  continue;
521  }
522 
523  // Get the buffer type from the bufferized function type.
524  Type memrefType = bufferizedFuncType.getInput(idx);
525  Value memref = frontBlock.addArgument(memrefType, bbArg.getLoc());
526  OpBuilder b(funcOp->getContext());
527  b.setInsertionPointToStart(&frontBlock);
528  // Replace all uses of bbArg through a ToMemRefOp.
529  for (auto &use : llvm::make_early_inc_range(bbArg.getUses())) {
530  if (auto toMemrefOp =
531  dyn_cast<bufferization::ToMemrefOp>(use.getOwner())) {
532  if (memref.getType() != toMemrefOp.memref().getType()) {
533  // Type has changed, insert a cast.
534  assert(memref::CastOp::areCastCompatible(
535  memref.getType(), toMemrefOp.memref().getType()) &&
536  "bufferizeFuncOpBoundary: cast incompatible");
537  auto castOp = b.create<memref::CastOp>(
538  funcOp.getLoc(), toMemrefOp.memref().getType(), memref);
539  toMemrefOp.memref().replaceAllUsesWith(castOp);
540  } else {
541  // Type did not change, replace directly.
542  toMemrefOp.memref().replaceAllUsesWith(memref);
543  }
544  }
545  }
546  // Replace all remaining uses by a to_tensor.
547  if (!bbArg.use_empty()) {
548  auto toTensorOp =
549  b.create<bufferization::ToTensorOp>(funcOp.getLoc(), memref);
550  bbArg.replaceAllUsesWith(toTensorOp);
551  }
552  frontBlock.eraseArgument(0);
553  // TODO: add support to erase aliasInfo entries if deemed necessary.
554  }
555 
556  // 4. Rewrite the FuncOp type to buffer form.
557  funcOp.setType(bufferizedFuncType);
558 
559  return success();
560 }

570  FuncCallerMap &callerMap) {
571  // For each FuncOp, the set of functions called by it (i.e. the union of
572  // symbols of all nested CallOpInterfaceOp).
573  DenseMap<func::FuncOp, DenseSet<func::FuncOp>> calledBy;
574  // For each FuncOp, the number of CallOpInterface it contains.
575  DenseMap<func::FuncOp, unsigned> numberCallOpsContainedInFuncOp;
576  WalkResult res = moduleOp.walk([&](func::FuncOp funcOp) -> WalkResult {
577  if (!funcOp.getBody().empty()) {
578  func::ReturnOp returnOp = getAssumedUniqueReturnOp(funcOp);
579  if (!returnOp)
580  return funcOp->emitError()

582  "without a unique ReturnOp";
583  }
584 
585  numberCallOpsContainedInFuncOp[funcOp] = 0;
586  return funcOp.walk([&](CallOpInterface callOp) -> WalkResult {
587  // Only support CallOp for now.
588  if (!isa<func::CallOp>(callOp.getOperation()))
589  return callOp->emitError() << "expected a CallOp";
590  func::FuncOp calledFunction = getCalledFunction(callOp);
591  assert(calledFunction && "could not retrieved called func::FuncOp");
592  auto it = callerMap.try_emplace(calledFunction, DenseSet<Operation *>{});
593  it.first->getSecond().insert(callOp);
594  if (calledBy[calledFunction].count(funcOp) == 0) {
595  calledBy[calledFunction].insert(funcOp);
596  numberCallOpsContainedInFuncOp[funcOp]++;
597  }
598  return WalkResult::advance();
599  });
600  });
601  if (res.wasInterrupted())
602  return failure();
603  // Iteratively remove function operation that do not call any of the
604  // functions remaining in the callCounter map and add them to the worklist.
605  while (!numberCallOpsContainedInFuncOp.empty()) {
606  auto it = llvm::find_if(numberCallOpsContainedInFuncOp,
607  [](auto entry) { return entry.getSecond() == 0; });
608  if (it == numberCallOpsContainedInFuncOp.end())
609  return moduleOp.emitOpError(
610  "expected callgraph to be free of circular dependencies.");
611  orderedFuncOps.push_back(it->getFirst());
612  for (auto callee : calledBy[it->getFirst()])
613  numberCallOpsContainedInFuncOp[callee]--;
614  numberCallOpsContainedInFuncOp.erase(it);
615  }
616  return success();
617 }

629 /// This is a purely mechanical process that may later become part of a
630 /// separate pass with its own layout assignment heuristic.
631 static void layoutPostProcessing(ModuleOp moduleOp) {
632  SmallVector<func::FuncOp> orderedFuncOps;
633  DenseMap<func::FuncOp, DenseSet<Operation *>> callerMap;
634  auto res = getFuncOpsOrderedByCalls(moduleOp, orderedFuncOps, callerMap);
635  (void)res;
636  assert(succeeded(res) && "unexpected getFuncOpsOrderedByCalls failure");
637 
638  for (func::FuncOp funcOp : orderedFuncOps) {
639  DenseMap<Operation *, SmallVector<Value>> operandsPerCaller;
640  foreachCaller(callerMap, funcOp, [&](Operation *caller) {
641  operandsPerCaller.try_emplace(caller, SmallVector<Value>());
642  });
643 
644  SmallVector<Type> argumentTypes;
645  // Iterate on each function argument and check it it was marked with a
646  // desired layout.
647  for (const auto &it :
648  llvm::enumerate(funcOp.getFunctionType().getInputs())) {
649  int argNumber = it.index();
650  Type inputType = it.value();
651  auto memrefType = inputType.dyn_cast<MemRefType>();
652  auto layoutAttr = funcOp.getArgAttrOfType<AffineMapAttr>(
653  argNumber, BufferizableOpInterface::kBufferLayoutAttrName);
654  AffineMap desiredLayoutMap =
655  layoutAttr ? layoutAttr.getValue() : AffineMap();
656  AffineMap currentLayoutMap =
657  memrefType ? getStridedLinearLayoutMap(memrefType) : AffineMap();
658  if (!memrefType || !layoutAttr || desiredLayoutMap == currentLayoutMap) {
659  argumentTypes.push_back(inputType);
660  foreachCaller(callerMap, funcOp, [&](Operation *caller) {
661  operandsPerCaller.find(caller)->getSecond().push_back(
662  caller->getOperand(argNumber));
663  });
664  continue;
665  }
666 
667  // Compute the buffer type with desired layout and add to input argument
668  // types.
669  MemRefType desiredMemrefType = MemRefType::get(
670  memrefType.getShape(), memrefType.getElementType(), desiredLayoutMap);
671  argumentTypes.push_back(desiredMemrefType);
672 
673  // If funcOp's body is not empty, change the bbArg type and propagate.
674  if (!funcOp.getBody().empty()) {
675  BlockArgument bbArg = funcOp.getArgument(argNumber);
676  bbArg.setType(desiredMemrefType);
677  OpBuilder b(bbArg.getContext());
678  b.setInsertionPointToStart(bbArg.getOwner());
679  assert(memref::CastOp::areCastCompatible(bbArg.getType(), memrefType) &&
680  "layoutPostProcessing: cast incompatible");
681  // Cast back to the original memrefType and let it canonicalize.
682  Value cast =
683  b.create<memref::CastOp>(funcOp.getLoc(), memrefType, bbArg);
684  bbArg.replaceAllUsesExcept(cast, cast.getDefiningOp());
685  }
686 
687  // Cast to desired buffer type on all callers to `funcOp`.
688  // TODO: on the callee side, this may even have to trigger a copy to
689  // change the layout. For now let the memref::CastOp fail to verify in
690  // such cases.
691  auto castArg = [&](Operation *caller) {
692  OpBuilder b(caller);
693  assert(
694  memref::CastOp::areCastCompatible(
695  caller->getOperand(argNumber).getType(), desiredMemrefType) &&
696  "layoutPostProcessing.2: cast incompatible");
697  Value newOperand = b.create<memref::CastOp>(
698  funcOp.getLoc(), desiredMemrefType, caller->getOperand(argNumber));
699  operandsPerCaller.find(caller)->getSecond().push_back(newOperand);
700  };
701  foreachCaller(callerMap, funcOp, castArg);
702  }
703 
704  // Set operands with cast buffer on all callers to `funcOp`.
705  foreachCaller(callerMap, funcOp, [&](Operation *caller) {
706  caller->setOperands(operandsPerCaller.lookup(caller));
707  });
708 
709  // Finally set the funcOp type to update the arguments.
710  auto newFuncType = FunctionType::get(moduleOp.getContext(), argumentTypes,
711  funcOp.getFunctionType().getResults());
712  funcOp.setType(newFuncType);
713  }

829  /// In a first approximation, all the function arguments of a func::FuncOp are
830  /// marked inplaceable. For now, it is the responsibility of the `callOp`
831  /// bufferization to allow func::FuncOp that are inplaceable to write inPlace.
832  LogicalResult bufferize(Operation *op, RewriterBase &rewriter,
833  BufferizationState &state) const {
834  func::CallOp callOp = cast<func::CallOp>(op);
835  unsigned numResults = callOp.getNumResults();
836  unsigned numOperands = callOp->getNumOperands();
837  func::FuncOp funcOp = getCalledFunction(callOp);
838  assert(funcOp && "expected CallOp to a func::FuncOp");
839  const FuncAnalysisState &funcState =
840  getFuncAnalysisState(state.getAnalysisState());
841  const OneShotBufferizationOptions &options =
842  static_cast<const OneShotBufferizationOptions &>(state.getOptions());
843 
844  // Result types of the bufferized CallOp.
845  SmallVector<Type> resultTypes;
846  // Replacement values for the existing CallOp. These are usually the results
847  // of the bufferized CallOp, unless a tensor result folds onto an operand.
848  SmallVector<Value> replacementValues(numResults, Value());
849  // For non-tensor results: A mapping from return val indices of the old
850  // CallOp to return val indices of the bufferized CallOp.
851  SmallVector<Optional<unsigned>> retValMapping(numResults, None);
852  // Operands of the bufferized CallOp.
853  SmallVector<Value> newOperands(numOperands, Value());
854 
855  // Based on previously gathered equivalence information, we know if a
856  // tensor result folds onto an operand. These are the only tensor value
857  // results that are supported at the moment.
858  //
859  // For tensors return values that do not fold onto an operand, additional
860  // work is needed (TODO) to either:
861  // * hoist a result into an inplaceable operand or
862  // * devise a better representation to truly return a buffer.
863  //
864  // Note: If a function has no body, no equivalence information is
865  // available. Consequently, a tensor return value cannot be proven to fold
866  // onto a func::FuncOp bbArg, so calls to such functions are not
867  // bufferizable at the moment.
868 
869  // 1. Compute the result types of the new CallOp. Tensor results that are
870  // equivalent to a func::FuncOp bbArg are no longer returned.
871  for (const auto &it : llvm::enumerate(callOp.getResultTypes())) {
872  unsigned returnValIdx = it.index();
873  Type returnType = it.value();
874  if (!isaTensor(returnType)) {
875  // Non-tensor values are returned.
876  retValMapping[returnValIdx] = resultTypes.size();
877  resultTypes.push_back(returnType);
878  continue;
879  }
880 
881  if (Optional<int64_t> bbArgIdx =
882  getEquivalentFuncArgIdx(funcOp, funcState, returnValIdx)) {
883  // Return operands that are equivalent to some bbArg, are not
884  // returned.
885  FailureOr<Value> bufferOrFailure =
886  state.getBuffer(rewriter, callOp->getOpOperand(*bbArgIdx));
887  if (failed(bufferOrFailure))
888  return failure();
889  replacementValues[returnValIdx] = *bufferOrFailure;
890  newOperands[*bbArgIdx] = *bufferOrFailure;
891  continue;
892  }
893 
894  if (!options.allowReturnAllocs)
895  return callOp->emitError(
896  "call to FuncOp that returns non-equivalent tensors not supported");
897 
898  // Returning a memref. This memref is not equivalent to any bbArg. It is
899  // likely a newly allocated buffer. We may want to hoist such allocations
900  // to the call site in the future.
901  retValMapping[returnValIdx] = resultTypes.size();
902  resultTypes.push_back(
903  funcOp.getFunctionType().getResult(resultTypes.size()));
904  }
905 
906  // 2. Compute bufferized FunctionType.
907  SmallVector<Type> argumentTypes{callOp->getOperandTypes()};
908  // Get the bufferized FunctionType for funcOp or construct it if not yet
909  // available.
910  FunctionType bufferizedFuncType = getBufferizedFunctionType(
911  funcOp.getContext(), argumentTypes, resultTypes, options);
912 
913  // 3. Rewrite tensor operands as memrefs based on `bufferizedFuncType`.
914  for (OpOperand &opOperand : callOp->getOpOperands()) {
915  unsigned idx = opOperand.getOperandNumber();
916  Value tensorOperand = opOperand.get();
917 
918  // Non-tensor operands are just copied.
919  if (!tensorOperand.getType().isa<TensorType>()) {
920  newOperands[idx] = tensorOperand;
921  continue;
922  }
923 
924  // Retrieve buffers for tensor operands. Tensor operand buffers, who's
925  // corresponding func::FuncOp bbArgs are equivalent to a returned tensor,
926  // were already stored in `newOperands` during Step 1.
927  Value buffer = newOperands[idx];
928  if (!buffer) {
929  FailureOr<Value> bufferOrFailure = state.getBuffer(rewriter, opOperand);
930  if (failed(bufferOrFailure))
931  return failure();
932  buffer = *bufferOrFailure;
933  }
934 
935  // Caller / callee type mismatch is handled with a CastOp.
936  auto memRefType = bufferizedFuncType.getInput(idx);
937  // Since we don't yet have a clear layout story, to_memref may
938  // conservatively turn tensors into more dynamic memref than necessary.
939  // If the memref type of the callee fails, introduce an extra memref.cast
940  // that will either canonicalize away or fail compilation until we can do
941  // something better.
942  if (buffer.getType() != memRefType) {
943  assert(
944  memref::CastOp::areCastCompatible(buffer.getType(), memRefType) &&
945  "CallOp::bufferize: cast incompatible");
946  Value castBuffer = rewriter.create<memref::CastOp>(callOp.getLoc(),
947  memRefType, buffer);
948  buffer = castBuffer;
949  }
950  newOperands[idx] = buffer;
951  }
952 
953  // 4. Create the new CallOp.
954  Operation *newCallOp = rewriter.create<func::CallOp>(
955  callOp.getLoc(), funcOp.getSymName(), resultTypes, newOperands);
956  newCallOp->setAttrs(callOp->getAttrs());
957  // Get replacement values for non-tensor / non-equivalent results.
958  for (unsigned i = 0; i < replacementValues.size(); ++i) {
959  if (replacementValues[i])
960  continue;
961  replacementValues[i] = newCallOp->getResult(*retValMapping[i]);
962  }
963 
964  // 5. Replace the old op with the new op.
965  replaceOpWithBufferizedValues(rewriter, callOp, replacementValues);
966 
967  return success();
968  }

1089  funcState.startFunctionAnalysis(funcOp);

1096  return failure();

1099  funcState.analyzedFuncOps[funcOp] = FuncOpAnalysisState::Analyzed;

1103  annotateOpsWithBufferizationMarkers(funcOp, analysisState);
1104  }

1107  return success();
