19 /// The state maintained across applications of various ops implementing the
20 /// TransformOpInterface. The operations implementing this interface and the
21 /// surrounding structure are referred to as transform IR. The operations to
22 /// which transformations apply are referred to as payload IR. The state thus
23 /// contains the mapping between values defined in the transform IR ops and
24 /// payload IR ops. It assumes that each value in the transform IR can be used
25 /// at most once (since transformations are likely to change the payload IR ops
26 /// the value corresponds to). Checks that transform IR values correspond to
27 /// disjoint sets of payload IR ops throughout the transformation.
28 ///
29 /// A reference to this class is passed as an argument to "apply" methods of the
30 /// transform op interface. Thus the "apply" method can call
31 /// `state.getPayloadOps( getSomeOperand() )` to obtain the list of operations
32 /// associated with its operand and subject to transformation. The method is
33 /// expected to populate the `TransformResults` class instance in order to
34 /// update the mapping. The `applyTransform` method takes care of propagating
35 /// the state of `TransformResults` into the instance of this class.
36 ///
37 /// When applying transform IR operations with regions, the client is expected
38 /// to create a RegionScope RAII object to create a new "stack frame" for
39 /// values defined inside the region. The mappings from and to these values will
40 /// be automatically dropped when the object goes out of scope, typically at the
41 /// end of the "apply" function of the parent operation. If a region contains
42 /// blocks with arguments, the client can map those arguments to payload IR ops
43 /// using "mapBlockArguments".
44 class TransformState {
45  /// Mapping between a Value in the transform IR and the corresponding set of
46  /// operations in the payload IR.
47  using TransformOpMapping = DenseMap<Value, SmallVector<Operation *>>;
48 
49  /// Mapping between a payload IR operation and the transform IR value it is
50  /// currently associated with.
51  using TransformOpReverseMapping = DenseMap<Operation *, Value>;

53  /// Bidirectional mappings between transform IR values and payload IR
54  /// operations.
55  struct Mappings {
56  TransformOpMapping direct;
57  TransformOpReverseMapping reverse;
58  };

60 public:
61  /// Creates a state for transform ops living in the given region. The parent
62  /// operation of the region. The second argument points to the root operation
63  /// in the payload IR beind transformed, which may or may not contain the
64  /// region with transform ops.
65  TransformState(Region &region, Operation *root);

67  /// Returns the op at which the transformation state is rooted. This is
68  /// typically helpful for transformations that apply globally.
69  Operation *getTopLevel() const;
70 
71  /// Returns the list of ops that the given transform IR value corresponds to.
72  /// This is helpful for transformations that apply to a particular handle.
73  ArrayRef<Operation *> getPayloadOps(Value value) const;

75  /// Applies the transformation specified by the given transform op and updates
76  /// the state accordingly.

79  /// Records the mapping between a block argument in the transform IR and a
80  /// list of operations in the payload IR. The arguments must be defined in
81  /// blocks of the currently processed transform IR region, typically after a
82  /// region scope is defined.
83  LogicalResult mapBlockArguments(BlockArgument argument,
84  ArrayRef<Operation *> operations) {
85 #if LLVM_ENABLE_ABI_BREAKING_CHECKS
86  assert(argument.getParentRegion() == regionStack.back() &&
87  "mapping block arguments from a region other than the active one");
88 #endif // LLVM_ENABLE_ABI_BREAKING_CHECKS
89  return setPayloadOps(argument, operations);

95  /// Creates a new region scope for the given region. The region is expected to
96  /// be nested in the currently processed region.
97  // Implementation note: this method is inline but implemented outside of the
98  // class body to comply with visibility and full-declaration requirements.
99  inline RegionScope make_region_scope(Region &region);
100 
101  /// A RAII object maintaining a "stack frame" for a transform IR region. When
102  /// applying a transform IR operation that contains a region, the caller is
103  /// expected to create a RegionScope before applying the ops contained in the
104  /// region. This ensures that the mappings between values defined in the
105  /// transform IR region and payload IR operations are cleared when the region
106  /// processing ends; such values cannot be accessed outside the region.
107  class RegionScope {
108  public:
109  /// Forgets the mapping from or to values defined in the associated
110  /// transform IR region.
111  ~RegionScope() {
112  state.mappings.erase(region);
113 #if LLVM_ENABLE_ABI_BREAKING_CHECKS
114  state.regionStack.pop_back();
115 #endif // LLVM_ENABLE_ABI_BREAKING_CHECKS
116  }
117 
118  private:
119  /// Creates a new scope for mappings between values defined in the given
120  /// transform IR region and payload IR operations.
121  RegionScope(TransformState &state, Region &region)
122  : state(state), region(&region) {
123  auto res = state.mappings.try_emplace(this->region);
124  assert(res.second && "the region scope is already present");
125  (void)res;
126 #if LLVM_ENABLE_ABI_BREAKING_CHECKS
127  assert(state.regionStack.back()->isProperAncestor(&region) &&
128  "scope started at a non-nested region");
129  state.regionStack.push_back(&region);
130 #endif // LLVM_ENABLE_ABI_BREAKING_CHECKS
131  }
132 
133  /// Back-reference to the transform state.
134  TransformState &state;
135 
136  /// The region this scope is associated with.
137  Region *region;

169  /// Sets the payload IR ops associated with the given transform IR value.
170  /// Fails if this would result in multiple transform IR values with uses
171  /// corresponding to the same payload IR ops. For example, a hypothetical
172  /// "find function by name" transform op would (indirectly) call this
173  /// function for its result. Having two such calls in a row with for different
174  /// values, e.g. coming from different ops:
175  ///
176  /// %0 = transform.find_func_by_name { name = "myfunc" }
177  /// %1 = transform.find_func_by_name { name = "myfunc" }
178  ///
179  /// would lead to both values pointing to the same operation. The second call
180  /// to setPayloadOps will fail, unless the association with the %0 value is
181  /// removed first by calling update/removePayloadOps.
182  LogicalResult setPayloadOps(Value value, ArrayRef<Operation *> targets);
183 
184  /// Forgets the payload IR ops associated with the given transform IR value.
185  void removePayloadOps(Value value);
186 
187  /// Updates the payload IR ops associated with the given transform IR value.
188  /// The callback function is called once per associated operation and is
189  /// expected to return the modified operation or nullptr. In the latter case,
190  /// the corresponding operation is no longer associated with the transform IR
191  /// value.
192  void updatePayloadOps(Value value,
193  function_ref<Operation *(Operation *)> callback);
194 
195  /// The mappings between transform IR values and payload IR ops, aggregated by
196  /// the region in which the transform IR values are defined.
197  llvm::SmallDenseMap<Region *, Mappings> mappings;
198 
199  /// The top-level operation that contains all payload IR, typically a module.
200  Operation *topLevel;

204  /// Each region must be an ancestor of the following regions in this list.
205  /// These are also the keys for "mappings".
206  SmallVector<Region *> regionStack;
