83 void LinalgComprehensiveModuleBufferize::runOnOperation() {
84  OneShotBufferizationOptions opt;
85  if (!options) {
86  // Make new bufferization options if none were provided when creating the
87  // pass.
88  if (useAlloca) {
89  opt.allocationFn = allocationFnUsingAlloca;
90  opt.deallocationFn = [](OpBuilder &b, Location loc, Value v) {
91  return success();
92  };
93  }
94  opt.allowReturnAllocs = allowReturnAllocs;
95  opt.allowUnknownOps = allowUnknownOps;
96  opt.analysisFuzzerSeed = analysisFuzzerSeed;
97  opt.createDeallocs = createDeallocs;
98  opt.fullyDynamicLayoutMaps = fullyDynamicLayoutMaps;
99  opt.printConflicts = printConflicts;
100  opt.testAnalysisOnly = testAnalysisOnly;
101  opt.alwaysAliasingWithDest = alwaysAliasingWithDest;
102  if (initTensorElimination) {
103  opt.addPostAnalysisStep(insertSliceAnchoredInitTensorEliminationStep);
104  }
105  } else {
106  opt = *options;
107  }
108 
109  ModuleOp moduleOp = getOperation();
110  applyEnablingTransformations(moduleOp);
111 
112  if (failed(runModuleBufferize(moduleOp, opt))) {
113  signalPassFailure();
114  return;
115  }
116 
117  if (opt.testAnalysisOnly)
118  return;
119 
120  OpPassManager cleanupPipeline("builtin.module");
121  cleanupPipeline.addPass(createCanonicalizerPass());
122  cleanupPipeline.addPass(createCSEPass());
123  cleanupPipeline.addPass(createLoopInvariantCodeMotionPass());
124  (void)runPipeline(cleanupPipeline, moduleOp);
125 }
