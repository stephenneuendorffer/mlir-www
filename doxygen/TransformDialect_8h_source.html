23 /// Asserts that the operations provided as template arguments implement the
24 /// TransformOpInterface. This must be a dynamic assertion since interface
25 /// implementations may be registered at runtime.
26 template <typename OpTy>
27 static inline void checkImplementsTransformInterface(MLIRContext *context) {
28  // Since the operation is being inserted into the Transform dialect and the
29  // dialect does not implement the interface fallback, only check for the op
30  // itself having the interface implementation.
31  RegisteredOperationName opName =
32  *RegisteredOperationName::lookup(OpTy::getOperationName(), context);
33  assert(opName.hasInterface<TransformOpInterface>() &&
34  "ops injected into the transform dialect must implement "
35  "TransformOpInterface");
36 }
37 } // namespace detail
38 #endif // NDEBUG

40 /// Base class for extensions of the Transform dialect that supports injecting
41 /// operations into the Transform dialect at load time. Concrete extensions are
42 /// expected to derive this class and register operations in the constructor.
43 /// They can be registered with the DialectRegistry and automatically applied
44 /// to the Transform dialect when it is loaded.
45 template <typename DerivedTy, typename... ExtraDialects>
46 class TransformDialectExtension
47  : public DialectExtension<DerivedTy, TransformDialect, ExtraDialects...> {
48  using Initializer = std::function<void(TransformDialect *)>;
49  using DialectLoader = std::function<void(MLIRContext *)>;
50 
51 public:
52  /// Extension application hook. Actually loads the dependent dialects and
53  /// registers the additional operations. Not expected to be called directly.
54  void apply(MLIRContext *context, TransformDialect *transformDialect,
55  ExtraDialects *...) const final {
56  for (const DialectLoader &loader : dialectLoaders)
57  loader(context);
58  for (const Initializer &init : opInitializers)
59  init(transformDialect);
60  }
61 
62 protected:
63  /// Injects the operations into the Transform dialect. The operations must
64  /// implement the TransformOpInterface and the implementation must be already
65  /// available when the operation is injected.
66  template <typename... OpTys>
67  void registerTransformOps() {
68  opInitializers.push_back([](TransformDialect *transformDialect) {
69  transformDialect->addOperations<OpTys...>();
70 
71 #ifndef NDEBUG
72  (void)std::initializer_list<int>{
73  (detail::checkImplementsTransformInterface<OpTys>(
74  transformDialect->getContext()),
75  0)...};
76 #endif // NDEBUG
77  });
78  }
79 
80  /// Declares that this Transform dialect extension depends on the dialect
81  /// provided as template parameter. When the Transform dialect is loaded,
82  /// dependent dialects will be loaded as well. This is intended for dialects
83  /// that contain attributes and types used in creation and canonicalization of
84  /// the injected operations.
85  template <typename DialectTy>
86  void declareDependentDialect() {
87  dialectLoaders.push_back(
88  [](MLIRContext *context) { context->loadDialect<DialectTy>(); });
89  }
90 
91 private:
92  SmallVector<Initializer> opInitializers;
93  SmallVector<DialectLoader> dialectLoaders;
94 };
